```typescript
import { prisma } from '../../lib/prisma'

// GIF 1x1 transparente (35 bytes)
export const TRANSPARENT_GIF = Buffer.from(
  'R0lGODlhAQABAIAAAAAAAP///yH5BAEAAAAALAAAAAABAAEAAAIBRAA7',
  'base64'
)

export async function recordOpen(recipientId: string): Promise<void> {
  const recipient = await prisma.campaignRecipient.findUnique({
    where: { id: recipientId },
    select: { id: true, status: true }
  })
  if (!recipient || recipient.status === 'OPENED' || recipient.status === 'CLICKED') return
  await prisma.campaignRecipient.update({
    where: { id: recipientId },
    data: { status: 'OPENED', openedAt: new Date() }
  })
}

export async function recordClick(recipientId: string): Promise<void> {
  const recipient = await prisma.campaignRecipient.findUnique({
    where: { id: recipientId },
    select: { id: true, status: true }
  })
  if (!recipient || recipient.status === 'CLICKED') return
  await prisma.campaignRecipient.update({
    where: { id: recipientId },
    data: { status: 'CLICKED', clickedAt: new Date() }
  })
}
```

```typescript
// Backend/src/modules/tracking/tracking.routes.ts
import { Router, Request, Response } from 'express'
import { recordOpen, recordClick, TRANSPARENT_GIF } from './tracking.service'

export const trackingRoutes = Router()

trackingRoutes.get('/o/:recipientId', async (req: Request, res: Response) => {
  // Fire-and-forget — always return GIF to avoid leaking info
  recordOpen(req.params.recipientId).catch(() => {})
  res.set({
    'Content-Type': 'image/gif',
    'Content-Length': TRANSPARENT_GIF.length,
    'Cache-Control': 'no-store, no-cache, must-revalidate',
    'Pragma': 'no-cache'
  })
  res.send(TRANSPARENT_GIF)
})

trackingRoutes.get('/c/:recipientId', async (req: Request, res: Response) => {
  const rawUrl = req.query.url as string | undefined
  if (!rawUrl) return res.status(400).json({ error: 'Missing url param' })

  let decodedUrl: string
  try {
    decodedUrl = decodeURIComponent(rawUrl)
    const parsed = new URL(decodedUrl)
    if (parsed.protocol !== 'http:' && parsed.protocol !== 'https:') throw new Error()
  } catch {
    return res.status(400).json({ error: 'Invalid url' })
  }

  recordClick(req.params.recipientId).catch(() => {})
  res.redirect(302, decodedUrl)
})
```

```typescript
export function injectTrackingIntoHtml(html: string, recipientId: string, baseUrl: string): string {
  // Reescribir hrefs (excepto mailto/tel/anchors)
  let result = html.replace(
    /href="(https?:\/\/[^"]+)"/gi,
    (_, url) => `href="${baseUrl}/t/c/${recipientId}?url=${encodeURIComponent(url)}"`
  )

  // Pixel de apertura
  const pixel = `<img src="${baseUrl}/t/o/${recipientId}" width="1" height="1" style="display:none" alt="" />`

  if (result.includes('</body>')) {
    result = result.replace('</body>', `${pixel}</body>`)
  } else {
    result += pixel
  }

  return result
}
```

```typescript
// GET /api/campaigns/:id/stats
router.get('/:id/stats', authenticate, async (req, res) => {
  const { workspaceId } = req.user!
  const campaign = await prisma.campaign.findFirst({
    where: { id: req.params.id, workspaceId }
  })
  if (!campaign) return res.status(404).json({ error: 'Not found' })

  const counts = await prisma.campaignRecipient.groupBy({
    by: ['status'],
    where: { campaignId: campaign.id, workspaceId },
    _count: { status: true }
  })

  const byStatus = Object.fromEntries(counts.map(c => [c.status, c._count.status]))
  const total = Object.values(byStatus).reduce((a, b) => a + b, 0)
  const sent = (byStatus['SENT'] ?? 0) + (byStatus['OPENED'] ?? 0) + (byStatus['CLICKED'] ?? 0)
  const failed = byStatus['FAILED'] ?? 0
  const opened = (byStatus['OPENED'] ?? 0) + (byStatus['CLICKED'] ?? 0)
  const clicked = byStatus['CLICKED'] ?? 0

  res.json({
    total,
    sent,
    failed,
    opened,
    clicked,
    openRate: sent > 0 ? opened / sent : 0,
    clickRate: sent > 0 ? clicked / sent : 0
  })
})
```

```typescript
export async function sendToSingleContact(params: {
  campaignId: string
  contactId: string
  workspaceId: string
}): Promise<void> {
  const { campaignId, contactId, workspaceId } = params

  const [campaign, contact] = await Promise.all([
    prisma.campaign.findFirst({ where: { id: campaignId, workspaceId } }),
    prisma.contact.findUnique({ where: { id: contactId } })
  ])

  if (!campaign || !contact) return

  // Comprobar suppression
  const suppressed = await prisma.suppression.findFirst({
    where: { workspaceId, OR: [{ value: contact.email ?? '' }, { value: contact.phone ?? '' }] }
  })
  if (suppressed) return

  const recipient = await prisma.campaignRecipient.create({
    data: { campaignId, workspaceId, contactId, status: 'PENDING', createdAt: new Date() }
  })

  const driver = getDriver() // función existente que retorna el driver activo
  let renderedBody = renderMergeTags(campaign.body ?? '', contact)

  if (campaign.channel === 'EMAIL') {
    renderedBody = injectTrackingIntoHtml(renderedBody, recipient.id, process.env.API_BASE_URL ?? '')
  }

  try {
    if (campaign.channel === 'EMAIL') {
      await driver.sendEmail(contact.email!, campaign.subject ?? '', renderedBody)
    } else if (campaign.channel === 'SMS') {
      await driver.sendSms(contact.phone!, renderedBody)
    } else if (campaign.channel === 'WHATSAPP') {
      await driver.sendWhatsapp(contact.phone!, renderedBody)
    }
    await prisma.campaignRecipient.update({
      where: { id: recipient.id },
      data: { status: 'SENT', sentAt: new Date() }
    })
  } catch (err) {
    await prisma.campaignRecipient.update({
      where: { id: recipient.id },
      data: { status: 'FAILED', error: String(err) }
    })
  }
}
```

```typescript
  if (!field.conditions?.length) return true
```
