**Architecture:** Tres endpoints públicos (sin auth) se registran en app.ts: un webhook de email (Resend/Postmark dual-shape), un webhook SMS (Twilio con HMAC-SHA1), y un endpoint de unsubscribe (HMAC-SHA256 sobre recipientId). **Tech Stack:** TypeScript, Express 4, Prisma 5 (PostgreSQL), Node.js `crypto` built-in, Twilio (webhook validation), Resend/Postmark (email inbound)

## Global Constraints

- Webhook routes: sin `authenticate` middleware — servicios externos no pueden enviar JWT
- TWILIO_AUTH_TOKEN: env var, requerido para validación HMAC-SHA1 de SMS
- JWT_SECRET: reutilizado para HMAC-SHA256 de unsubscribe token
- API_BASE_URL: env var para construir URLs de unsubscribe (ya existe desde 3A)
- timingSafeEqual: OBLIGATORIO para comparar tokens/firmas (previene timing attacks)
- Unsubscribe: idempotente — re-click mismo link = 200 HTML, sin error
- SMS inbound: responder con `<Response/>` TwiML vacío (no auto-reply)
- Toda query DB con `workspaceId` scope

// Backend/src/lib/unsubscribe.ts
import { createHmac, timingSafeEqual } from 'crypto'

export function generateUnsubscribeToken(recipientId: string): string {
  const secret = process.env.JWT_SECRET ?? ''
  return createHmac('sha256', secret).update(recipientId).digest('hex')
}

export function verifyUnsubscribeToken(recipientId: string, token: string): boolean {
  if (!token) return false
  const expected = generateUnsubscribeToken(recipientId)
  try {
    const a = Buffer.from(expected, 'hex')
    const b = Buffer.from(token, 'hex')
    if (a.length !== b.length) return false
    return timingSafeEqual(a, b)
  } catch {
    return false
  }
}
```

import { verifyUnsubscribeToken } from '../../lib/unsubscribe'

// Añadir ruta:
trackingRoutes.get('/u/:recipientId/:token', async (req: Request, res: Response) => {
  const { recipientId, token } = req.params

  if (!verifyUnsubscribeToken(recipientId, token)) {
    return res.status(400).send('Enlace de baja inválido o expirado.')
  }

  const recipient = await prisma.campaignRecipient.findUnique({
    where: { id: recipientId },
    include: { contact: { select: { email: true, phone: true } } }
  })

  if (!recipient) {
    // Devolver HTML de éxito de todas formas (no revelar si existe)
    return res.status(200).send(unsubscribeHtml())
  }

  if (recipient.status !== 'UNSUBSCRIBED') {
    await prisma.campaignRecipient.update({
      where: { id: recipientId },
      data: { status: 'UNSUBSCRIBED' }
    })

    // Upsert en Suppression por email y/o phone
    const contact = (recipient as any).contact
    if (contact?.email) {
      await prisma.suppression.upsert({
        where: { workspaceId_value: { workspaceId: recipient.workspaceId, value: contact.email } },
        create: { workspaceId: recipient.workspaceId, value: contact.email, type: 'EMAIL' },
        update: {}
      })
    }
    if (contact?.phone) {
      await prisma.suppression.upsert({
        where: { workspaceId_value: { workspaceId: recipient.workspaceId, value: contact.phone } },
        create: { workspaceId: recipient.workspaceId, value: contact.phone, type: 'SMS' },
        update: {}
      })
    }
  }

  res.status(200).send(unsubscribeHtml())
})

function unsubscribeHtml(): string {
  return `<!DOCTYPE html>

</head>
<body><div class="card"><h1>✓ Te has dado de baja</h1>
<p>No recibirás más mensajes de este remitente.</p></div></body></html>`
}

export function injectTrackingIntoHtml(html: string, recipientId: string, baseUrl: string): string {
  // Reescribir hrefs (excepto mailto/tel/anchors)
  let result = html.replace(
    /href="(https?:\/\/[^"]+)"/gi,
    (_, url) => `href="${baseUrl}/t/c/${recipientId}?url=${encodeURIComponent(url)}"`
  )

  // Pixel de apertura
  const pixel = `<img src="${baseUrl}/t/o/${recipientId}" width="1" height="1" style="display:none" alt="" />`

  // Footer de unsubscribe
  const unsubToken = generateUnsubscribeToken(recipientId)
  const unsubUrl = `${baseUrl}/t/u/${recipientId}/${unsubToken}`
  const unsubFooter = `<p style="text-align:center;color:#9ca3af;font-size:11px;margin-top:32px">
<a href="${unsubUrl}" style="color:#9ca3af">Darse de baja</a>
</p>`

  const inject = `${pixel}${unsubFooter}`

  if (result.includes('</body>')) {
    result = result.replace('</body>', `${inject}</body>`)
  } else {
    result += inject
  }

  return result
}
```

  const sortedParams = Object.keys(params).sort()

// Backend/src/modules/webhooks/sms-inbound.routes.ts
import { Router, Request, Response } from 'express'
import { createHmac, timingSafeEqual } from 'crypto'
import { processInboundMessage } from '../messaging/message.service'

export const smsInboundRoutes = Router()

function validateTwilioSignature(req: Request): boolean {
  const authToken = process.env.TWILIO_AUTH_TOKEN
  if (!authToken) return false

  const signature = req.headers['x-twilio-signature'] as string | undefined
  if (!signature) return false

  // Reconstruir la URL completa que Twilio usó para firmar
  const proto = req.headers['x-forwarded-proto'] ?? req.protocol
  const host = req.headers['x-forwarded-host'] ?? req.headers.host
  const url = `${proto}://${host}${req.originalUrl}`

  // Ordenar params del body alfabéticamente y concatenar
  const params = req.body as Record<string, string>
  const sortedKeys = Object.keys(params).sort()
  let str = url
  for (const key of sortedKeys) {
    str += key + (params[key] ?? '')
  }

  const expected = createHmac('sha1', authToken).update(str).digest('base64')

  try {
    const a = Buffer.from(expected, 'base64')
    const b = Buffer.from(signature, 'base64')
    if (a.length !== b.length) return false
    return timingSafeEqual(a, b)
  } catch {
    return false
  }
}

smsInboundRoutes.post('/sms-inbound', async (req: Request, res: Response) => {
  if (!validateTwilioSignature(req)) {
    return res.status(403).send('Forbidden')
  }

  const params = req.body as Record<string, string>
  const from = params.From
  const body = params.Body
  const to = params.To

  if (!from || !body) {
    return res.status(200).type('text/xml').send('<Response/>')  // TwiML vacío siempre
  }

  const workspaceId = resolveWorkspaceFromPhone(to)
  if (workspaceId) {
    processInboundMessage({ workspaceId, from, body, channel: 'SMS' }).catch(console.error)
  }

  // Siempre responder con TwiML vacío (no auto-reply)
  res.status(200).type('text/xml').send('<Response/>')
})

function resolveWorkspaceFromPhone(toPhone: string): string | null {
  // Opción 1: Match por TWILIO_FROM_NUMBER env
  const configured = process.env.TWILIO_FROM_NUMBER
  if (configured && configured === toPhone) {
    return process.env.DEFAULT_WORKSPACE_ID ?? null
  }
  // Opción 2: Buscar en DB — implementar si hay múltiples workspaces con teléfonos distintos
  return process.env.DEFAULT_WORKSPACE_ID ?? null
}
